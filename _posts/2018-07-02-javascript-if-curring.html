//change curring
function curringOut(argApiName) 
{
    var apiName = argApiName;
    return function(condition) 
    {
         if(!condition) return false;
         return apiName && condition;
    }
}

function findWord(word) 
{
    return askedFields.indexOf(word) > -1;
}
